 26 -def newtonRaphson(function, lox, hix, xacc): 

 27       
 28      """ 
 29      X{Newton-Raphson} algorithm for X{root} finding. 
 30      The algorithm used is a safe version of Newton-Raphson 
 31      (see page 366 of Numerical Recipes in C, 2ed). 
 32       
 33      @param function: function of one numerical variable that 
 34          uses only those operations that are defined for  DerivVar 
 35          objects in the module L{Scientific.Functions.FirstDerivatives} 
 36      @type function: callable 
 37      @param lox: lower limit of the search interval 
 38      @type lox: C{float} 
 39      @param hix: upper limit of the search interval 
 40      @type hix: C[{loat} 
 41      @param xacc: requested absolute precision of the root 
 42      @type xacc: C{float} 
 43      @returns: a root of function between lox and hix 
 44      @rtype:  C{float} 
 45      @raises ValueError: if C{function(lox)} and C{function(hix)} have 
 46          the same sign, or if there is no convergence after 500 iterations 
 47      """ 
 48   
 49      maxit = 500 
 50      tmp = function(DerivVar(lox)) 
 51      fl = tmp[0] 
 52      tmp = function(DerivVar(hix)) 
 53      fh = tmp[0] 
 54      if ((fl > 0.0 and fh > 0.0) or (fl < 0.0 and fh < 0.0)): 
 55          raise ValueError("Root must be bracketed") 
 56      if (fl == 0.0): return lox 
 57      if (fh == 0.0): return hix 
 58      if (fl < 0.0): 
 59          xl=lox 
 60          xh=hix 
 61      else: 
 62          xh=lox 
 63          xl=hix 
 64      rts=0.5*(lox+hix) 
 65      dxold=abs(hix-lox) 
 66      dx=dxold 
 67      tmp = function(DerivVar(rts)) 
 68      f = tmp[0] 
 69      df = tmp[1][0] 
 70      for j in range(maxit): 
 71          if ((((rts-xh)*df-f)*((rts-xl)*df-f) > 0.0) 
 72              or (abs(2.0*f) > abs(dxold*df))): 
 73              dxold=dx 
 74              dx=0.5*(xh-xl) 
 75              rts=xl+dx 
 76              if (xl == rts): return rts 
 77          else: 
 78              dxold=dx 
 79              dx=f/df 
 80              temp=rts 
 81              rts=rts-dx 
 82              if (temp == rts): return rts 
 83          if (abs(dx) < xacc): return rts 
 84          tmp = function(DerivVar(rts)) 
 85          f = tmp[0] 
 86          df = tmp[1][0] 
 87          if (f < 0.0): 
 88              xl=rts 
 89          else: 
 90              xh=rts 
 91      raise ValueError("Maximum number of iterations exceeded")
